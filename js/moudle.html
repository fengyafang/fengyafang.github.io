      # export 的输出及引入

      在a文件中export

      export function foo(){}
export function bar(){}

      在b文件中需要这样引入

      import {foo,bar} from './a' //名字必须一样

      如果这样引入会是undefind

      import serve from './a' //serve的值会是undefind

      想作为对象引入可以这样写

      import * as serve from './a'

      # export default 的输出及引入

      //a文件
 function foo(){}
 function bar(){}
 export default {
	 foo,
	 bar
 }

      export default引入

      //b文件
import serve from './a'

      这样是不行的

      import {foo,bar} from './a'  //foo,bar 并不是预想的值
